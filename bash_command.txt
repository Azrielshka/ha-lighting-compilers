# bash_command.txt — команды для тестирования/отладки ha-lighting-compilers
# Запускай из КОРНЯ проекта (там где лежат папки scripts/ и data/)

# ------------------------------------------------------------
# 0) БАЗОВАЯ ПРОВЕРКА ОКРУЖЕНИЯ
# ------------------------------------------------------------

# Где мы находимся + статус git
pwd
git status

# Проверка Python (должен быть .venv)
where python

# ------------------------------------------------------------
# 1) СТРУКТУРА ДАННЫХ
# ------------------------------------------------------------

# Быстро посмотреть структуру data/
ls -la data
ls -la data/normalized
ls -la data/light_groups

# ------------------------------------------------------------
# 2) NORMALIZE (две модели запуска)
# ------------------------------------------------------------

# 2.1 Через CLI (с явным Excel)
python scripts/normalize_excel.py --excel "data/Тестовая таблица.xlsx"

# 2.2 Через DEFAULT_EXCEL_PATH внутри скрипта
python scripts/normalize_excel.py

# Проверка, что parquet созданы
ls -la data/normalized

# ------------------------------------------------------------
# 3) ГЕНЕРАЦИЯ YAML ГРУПП
# ------------------------------------------------------------

# Подгруппы света
python scripts/generate_lights_groups.py
sed -n '1,80p' data/light_groups/lights_group.yaml

# Общие группы помещений
python scripts/generate_general_groups.py
sed -n '1,80p' data/light_groups/lights_general_groups.yaml

# Группы этажей
python scripts/generate_floor_groups.py
grep -n "этаж" data/light_groups/lights_floor_groups.yaml | head -n 20

# ------------------------------------------------------------
# 4) LOVELACE
# ------------------------------------------------------------

# Генерация карточек
python scripts/generate_lovelace_cards_v2.py

# Просмотр результата
sed -n '1,120p' data/lovelace_cards_generated.txt

# Найти конкретное помещение
grep -n "### 403" data/lovelace_cards_generated.txt | head -n 20

# ------------------------------------------------------------
# 5) FULL PIPELINE (CLI)
# ------------------------------------------------------------

# Полный пересбор
python scripts/normalize_excel.py --excel "data/Тестовая таблица.xlsx" && \
python scripts/generate_lights_groups.py && \
python scripts/generate_general_groups.py && \
python scripts/generate_floor_groups.py && \
python scripts/generate_lovelace_cards_v2.py

# ------------------------------------------------------------
# 6) LAUNCHER (DEV)
# ------------------------------------------------------------

# Запуск launcher из Python
python main.py

# ------------------------------------------------------------
# 7) EXE — НЕ СОБИРАЕМ РУКАМИ
# ------------------------------------------------------------
#
# EXE собирает GitHub Actions по тегу: .github/workflows/release.yml
#
#     git tag v3.0.1 && git push origin v3.0.1
#
# -> сборка на чистой windows-latest -> GitHub Release с ZIP.
#
# Правило: тег -> EXE, других EXE не бывает. Собранный локально приходит с
# чьей-то машины в неизвестном состоянии, и сказать, из чего он, нечем —
# а лаунчер показывает версию в шапке именно затем, чтобы на вопрос
# «что стоит на объекте» был ответ.
#
# Перед тегом:
#   1. прогнать pytest (CI их НЕ гоняет — решение владельца)
#   2. сверить launcher/__version__ с тегом. CI это проверит и упадёт при
#      расхождении, но лучше не гонять сборку впустую
#
# Собрать локально можно только для отладки САМОЙ сборки, не для раздачи:
#     python tools/make_icon.py build/launcher.ico
#     pyinstaller --onefile --noconsole --icon build/launcher.ico launcher/main.py --name launcher
