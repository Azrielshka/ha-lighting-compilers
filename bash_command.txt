# bash_command.txt — команды для тестирования/отладки ha-college-lighting
# Запускай из КОРНЯ проекта (там где лежат папки scripts/ и data/)

# ------------------------------------------------------------
# 0) БАЗОВАЯ ПРОВЕРКА ОКРУЖЕНИЯ
# ------------------------------------------------------------

# Где мы находимся + статус git
pwd
git status

# Проверка Python (должен быть .venv)
where python

# ------------------------------------------------------------
# 1) СТРУКТУРА ДАННЫХ
# ------------------------------------------------------------

# Быстро посмотреть структуру data/
ls -la data
ls -la data/normalized
ls -la data/light_groups

# ------------------------------------------------------------
# 2) NORMALIZE (две модели запуска)
# ------------------------------------------------------------

# 2.1 Через CLI (с явным Excel)
python scripts/normalize_excel.py --excel "data/Тестовая таблица.xlsx"

# 2.2 Через DEFAULT_EXCEL_PATH внутри скрипта
python scripts/normalize_excel.py

# Проверка, что parquet созданы
ls -la data/normalized

# ------------------------------------------------------------
# 3) ГЕНЕРАЦИЯ YAML ГРУПП
# ------------------------------------------------------------

# Подгруппы света
python scripts/generate_lights_groups.py
sed -n '1,80p' data/light_groups/lights_group.yaml

# Общие группы помещений
python scripts/generate_general_groups.py
sed -n '1,80p' data/light_groups/lights_general_groups.yaml

# Группы этажей
python scripts/generate_floor_groups.py
grep -n "этаж" data/light_groups/lights_floor_groups.yaml | head -n 20

# ------------------------------------------------------------
# 4) LOVELACE
# ------------------------------------------------------------

# Генерация карточек
python scripts/generate_lovelace_cards_v2.py

# Просмотр результата
sed -n '1,120p' data/lovelace_cards_generated.txt

# Найти конкретное помещение
grep -n "### 403" data/lovelace_cards_generated.txt | head -n 20

# ------------------------------------------------------------
# 5) FULL PIPELINE (CLI)
# ------------------------------------------------------------

# Полный пересбор
python scripts/normalize_excel.py --excel "data/Тестовая таблица.xlsx" && \
python scripts/generate_lights_groups.py && \
python scripts/generate_general_groups.py && \
python scripts/generate_floor_groups.py && \
python scripts/generate_lovelace_cards_v2.py

# ------------------------------------------------------------
# 6) LAUNCHER (DEV)
# ------------------------------------------------------------

# Запуск launcher из Python
python main.py

# ------------------------------------------------------------
# 7) BUILD EXE (PyInstaller)
# ------------------------------------------------------------

# Очистка старой сборки
rmdir /s /q build
rmdir /s /q dist

# Сборка exe
pyinstaller main.py --onefile --noconsole

# Запуск exe
dist\main.exe

# ------------------------------------------------------------
# 8) ОТЛАДКА EXE
# ------------------------------------------------------------

# Запуск exe с выводом в консоль (если нужно дебажить)
pyinstaller main.py --onefile

# ------------------------------------------------------------
# 9) GIT / DIFF
# ------------------------------------------------------------

# Посмотреть изменения
git diff

# Добавить изменения
git add .

# Коммит
git commit -m "wip: local changes"

# Пуш
git push

# ------------------------------------------------------------
# 10) УСТАНОВКА ЗАВИСИМОСТЕЙ
# ------------------------------------------------------------

pip install -r requirements.txt

# ------------------------------------------------------------
# 11) БЫСТРАЯ ДИАГНОСТИКА
# ------------------------------------------------------------

# Проверка наличия .venv
ls -la .venv

# Проверка python внутри venv
ls -la .venv/Scripts

# ------------------------------------------------------------
# ПРИМЕЧАНИЯ
# ------------------------------------------------------------

# 1. normalize_excel.py поддерживает:
#    - запуск с параметром --excel
#    - запуск без параметров (через DEFAULT_EXCEL_PATH)

# 2. Launcher использует:
#    <project_root>/.venv/Scripts/python.exe

# 3. Build All в UI = последовательный запуск всех скриптов

# 4. EXE может работать медленнее из-за распаковки PyInstaller
